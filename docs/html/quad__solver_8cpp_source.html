72void solve_quad(EqSolverData* data){
73 assert(data);
74 assert(EqSolverData::COEFF_NUM >= 3);
75
76 double a = data->coeffs[EqSolverData::COEFF_NUM - 3];
77 double b = data->coeffs[EqSolverData::COEFF_NUM - 2];
78 double c = data->coeffs[EqSolverData::COEFF_NUM - 1];
79
80 if (is_double_equal(a, 0)) { // check if equation is quad (zero division check)
81 solve_lin(data);
82 return;
83 }
84
85 double D = b*b - 4*a*c; // discriminant
86 if (is_double_equal(D, 0)) { // check if there are solutions
87 data->roots[0] = valid_root((-b) / (2*a));
88 data->roots_num = ONE_SOLUTION;
89 } else if (D < 0) {
90 data->roots_num = NO_SOLUTIONS;
91 } else {
92 double D_sqrt = NAN;
93 if (is_double_equal(c, 0))
94 D_sqrt = b;
95 else
96 D_sqrt = sqrt(D);
97 assert(isfinite(D_sqrt));
98
99 double double_a = a * 2;
100 data->roots[0] = valid_root((-b - D_sqrt) / double_a);
101 data->roots[1] = valid_root((-b + D_sqrt) / double_a);
102 data->roots_num = TWO_SOLUTIONS;
103 }
104 bubble_sort(data->roots, data->roots_num);
105}
